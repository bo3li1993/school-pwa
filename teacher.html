import { db } from './js/firebase-config.js'; import { collection, writeBatch, doc, addDoc, serverTimestamp } from 'https://www.gstatic.com/firebasejs/10.8.0/firebase-firestore.js'; window.saveAttendanceLive = async function() { const schoolId = window.getActiveSchoolId(); const currentUser = JSON.parse(localStorage.getItem('hs_user')); const todayISO = window.getTodayISO(); const batch = writeBatch(db); // تجميع الصفوف المعلمة غياب داخل الجدول const absentRows = document.querySelectorAll('.attendance-row[data-status="absent"]'); // 1. إجراء عملية الحفظ الأساسية للدفعة (Batch Commit) // (هنا كود الحفظ العادي الخاص بك لبيانات التحضير) try { await batch.commit(); // 2. ضخ الغائبين فوراً لـ notifications_queue ليعمل رادار الأخصائي والواتساب for(const row of absentRows) { const studentName = row.getAttribute('data-name'); const currentAbsencesCount = parseInt(row.getAttribute('data-total-absences') || '0') + 1; await addDoc(collection(db, 'notifications_queue'), { schoolId: schoolId, studentName: studentName, classId: selectedClassId, period: selectedPeriod, date: todayISO, message: `غياب الطالب: ${studentName} — فصل: ${selectedClassId} — الحصة: ${selectedPeriod} — تاريخ: ${todayISO}`, recordedBy: currentUser.name, type: 'absence', sent: false, createdAt: serverTimestamp() }); // 3. نظام الإنذارات الآلي التجاري: لو تجاوز الطالب 5 أيام غياب، يولد إنذاراً أحمر فورياً للأخصائي if(currentAbsencesCount >= 5) { await addDoc(collection(db, 'critical_warnings'), { schoolId: schoolId, studentName: studentName, classId: selectedClassId, totalDays: currentAbsencesCount, status: 'pending_warning_letter', createdAt: serverTimestamp() }); } } alert("✅ تم رصد حضور الصف، وضخ غيابات الطلاب لغرفة التدقيق الإداري بنجاح!"); } catch (error) { alert("❌ فشل عملية الحفظ السحابي: " + error.message); } };
